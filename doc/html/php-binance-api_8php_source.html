164  public function getProxyUriString()
165  {
166  $uri = isset( $this->proxyConf['proto'] ) ? $this->proxyConf['proto'] : "http";
167  // https://curl.haxx.se/libcurl/c/CURLOPT_PROXY.html
168  $supportedProxyProtocols = array('http', 'https', 'socks4', 'socks4a', 'socks5', 'socks5h');
169  if(in_array($uri, $supportedProxyProtocols) == false) {
170  die("Unknown proxy protocol '" . $this->proxyConf['proto'] . "', supported protocols are " . implode(", ",$supportedProxyProtocols) . "\n");
171  }
172 
173  $uri .= "://";
174  $uri .= isset( $this->proxyConf['address'] ) ? $this->proxyConf['address'] : "localhost";
175  if( isset( $this->proxyConf['address'] ) == false ) echo "warning: proxy address not set defaulting to localhost\n";
176  $uri .= ":";
177  $uri .= isset( $this->proxyConf['port'] ) ? $this->proxyConf['port'] : "1080";
178  if( isset( $this->proxyConf['address'] ) == false ) echo "warning: proxy port not set defaulting to 1080\n";
179  return $uri;

184  private function httpRequest($url, $method = "GET", $params = [], $signed = false) {
185  // is cURL installed yet?
186  if (!function_exists('curl_init')) {
187  die('Sorry cURL is not installed!');
188  }
189 
190  $ch = curl_init();
191  curl_setopt($ch, CURLOPT_VERBOSE, $this->httpDebug);
192  $query = http_build_query($params, '', '&');
193 
194  // signed with params
195  if($signed == true) {
196  if(empty($this->api_key) ) die("signedRequest error: API Key not set!");
197  if(empty($this->api_secret) ) die("signedRequest error: API Secret not set!");
198  $base = $this->base;
199  $ts = (microtime(true)*1000) + $this->info['timeOffset'];
200  $params['timestamp'] = number_format($ts,0,'.','');
201  if(isset($params['wapi'])) {
202  unset($params['wapi']);
203  $base = $this->wapi;
204  }
205  $query = http_build_query($params, '', '&');
206  $signature = hash_hmac('sha256', $query, $this->api_secret);
207  $endpoint = $base.$url.'?'.$query.'&signature='.$signature;
208  curl_setopt($ch, CURLOPT_URL, $endpoint);
209  curl_setopt($ch, CURLOPT_HTTPHEADER, array('X-MBX-APIKEY: ' . $this->api_key));
210  }
211  // params so buildquery string and append to url
212  else if(count($params)>0){
213  curl_setopt($ch, CURLOPT_URL, $this->base.$url.'?'.$query);
214  }
215  // no params so just the base url
216  else {
217  curl_setopt($ch, CURLOPT_URL, $this->base.$url);
218  curl_setopt($ch, CURLOPT_HTTPHEADER, array('X-MBX-APIKEY: ' . $this->api_key));
219  }
220 
221  curl_setopt($ch, CURLOPT_USERAGENT, "User-Agent: Mozilla/4.0 (compatible; PHP Binance API)");
222 
223  // Post and postfields
224  if($method == "POST") {
225  curl_setopt($ch, CURLOPT_POST, true);
226  //curl_setopt($ch, CURLOPT_POSTFIELDS, $query);
227  }
228 
229  // Delete Method
230  if($method == "DELETE") {
231  curl_setopt($ch, CURLOPT_CUSTOMREQUEST, $method);
232  }
233 
234  // proxy settings
235  if(is_array($this->proxyConf)) {
236  curl_setopt($ch, CURLOPT_PROXY, $this->getProxyUriString());
237  if(isset($this->proxyConf['user']) && isset($this->proxyConf['pass'])) {
238  curl_setopt($ch, CURLOPT_PROXYUSERPWD, $this->proxyConf['user'] . ':' . $this->proxyConf['pass']);
239  }
240  }
241 
242  curl_setopt($ch, CURLOPT_FOLLOWLOCATION, true);
243  // headers will proceed the output, json_decode will fail below
244  curl_setopt($ch, CURLOPT_HEADER, false);
245  curl_setopt($ch, CURLOPT_RETURNTRANSFER, true);
246  curl_setopt($ch, CURLOPT_TIMEOUT, 60);
247  $output = curl_exec($ch);
248 
249  // Check if any error occurred
250  if(curl_errno($ch) > 0) {
251  echo 'Curl error: ' . curl_error($ch) . "\n";
252  return [];
253  }
254 
255  curl_close($ch);
256  $json = json_decode($output, true);
257 
258  if(isset($json['msg'])) {
259  echo "signedRequest error: {$output}".PHP_EOL;
260  }
261 
262  $this->transfered += strlen( $output );
263  $this->requestCount++;
264 
265  return $json;
266  }

313  // Converts all your balances into a nice array
314  // If priceData is passed from $api->prices() it will add btcValue & btcTotal to each symbol
315  // This function sets $btc_value which is your estimated BTC value of all assets combined and $btc_total which includes amount on order
316  private function balanceData($array, $priceData = false) {
317  if ( $priceData ) $btc_value = $btc_total = 0.00;
318  $balances = [];
319  if ( empty($array) || empty($array['balances']) ) {
320  echo "balanceData error: Please make sure your system time is synchronized, or pass the useServerTime option.".PHP_EOL;
321  return [];
322  }
323  foreach ( $array['balances'] as $obj ) {
324  $asset = $obj['asset'];
325  $balances[$asset] = ["available"=>$obj['free'], "onOrder"=>$obj['locked'], "btcValue"=>0.00000000, "btcTotal"=>0.00000000];
326  if ( $priceData ) {
327  if ( $obj['free'] + $obj['locked'] < 0.00000001 ) continue;
328  if ( $asset == 'BTC' ) {
329  $balances[$asset]['btcValue'] = $obj['free'];
330  $balances[$asset]['btcTotal'] = $obj['free'] + $obj['locked'];
331  $btc_value+= $obj['free'];
332  $btc_total+= $obj['free'] + $obj['locked'];
333  continue;
334  }
335  $symbol = $asset.'BTC';
336  if ( $symbol == 'USDTBTC' ) {
337  $btcValue = number_format($obj['free'] / $priceData['BTCUSDT'],8,'.','');
338  $btcTotal = number_format(($obj['free'] + $obj['locked']) / $priceData['BTCUSDT'],8,'.','');
339  } elseif ( !isset($priceData[$symbol]) ) {
340  $btcValue = $btcTotal = 0;
341  } else {
342  $btcValue = number_format($obj['free'] * $priceData[$symbol],8,'.','');
343  $btcTotal = number_format(($obj['free'] + $obj['locked']) * $priceData[$symbol],8,'.','');
344  }
345  $balances[$asset]['btcValue'] = $btcValue;
346  $balances[$asset]['btcTotal'] = $btcTotal;
347  $btc_value+= $btcValue;
348  $btc_total+= $btcTotal;
349  }
350  }
351  if ( $priceData ) {
352  uasort($balances, function($a, $b) { return $a['btcValue'] < $b['btcValue']; });
353  $this->btc_value = $btc_value;
354  $this->btc_total = $btc_total;
355  }
356  return $balances;
357  }
