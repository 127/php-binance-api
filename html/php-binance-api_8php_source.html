820  private function httpRequest( string $url, string $method = "GET", array $params = [], bool $signed = false ) {
821  if( function_exists( 'curl_init' ) == false ) {
822  throw new \Exception( "Sorry cURL is not installed!" );
823  }
824 
825  $ch = curl_init();
826  curl_setopt( $ch, CURLOPT_VERBOSE, $this->httpDebug );
827  $query = http_build_query( $params, '', '&' );
828 
829  // signed with params
830  if( $signed == true ) {
831  if( empty( $this->api_key ) )
832  throw new \Exception( "signedRequest error: API Key not set!" );
833  if( empty( $this->api_secret ) )
834  throw new \Exception( "signedRequest error: API Secret not set!" );
835  $base = $this->base;
836  $ts = ( microtime( true ) * 1000 ) + $this->info[ 'timeOffset' ];
837  $params[ 'timestamp' ] = number_format( $ts, 0, '.', '' );
838  if( isset( $params[ 'wapi' ] ) ) {
839  unset( $params[ 'wapi' ] );
840  $base = $this->wapi;
841  }
842  $query = http_build_query( $params, '', '&' );
843  $signature = hash_hmac( 'sha256', $query, $this->api_secret );
844  $endpoint = $base . $url . '?' . $query . '&signature=' . $signature;
845  curl_setopt( $ch, CURLOPT_URL, $endpoint );
846  curl_setopt( $ch, CURLOPT_HTTPHEADER, array(
847  'X-MBX-APIKEY: ' . $this->api_key
848  ) );
849  }
850  // params so buildquery string and append to url
851  else if( count( $params ) > 0 ) {
852  curl_setopt( $ch, CURLOPT_URL, $this->base . $url . '?' . $query );
853  }
854  // no params so just the base url
855  else {
856  curl_setopt( $ch, CURLOPT_URL, $this->base . $url );
857  curl_setopt( $ch, CURLOPT_HTTPHEADER, array(
858  'X-MBX-APIKEY: ' . $this->api_key
859  ) );
860  }
861  curl_setopt( $ch, CURLOPT_USERAGENT, "User-Agent: Mozilla/4.0 (compatible; PHP Binance API)" );
862  // Post and postfields
863  if( $method == "POST" ) {
864  curl_setopt( $ch, CURLOPT_POST, true );
865  // curl_setopt($ch, CURLOPT_POSTFIELDS, $query);
866  }
867  // Delete Method
868  if( $method == "DELETE" ) {
869  curl_setopt( $ch, CURLOPT_CUSTOMREQUEST, $method );
870  }
871  // proxy settings
872  if( is_array( $this->proxyConf ) ) {
873  curl_setopt( $ch, CURLOPT_PROXY, $this->getProxyUriString() );
874  if( isset( $this->proxyConf[ 'user' ] ) && isset( $this->proxyConf[ 'pass' ] ) ) {
875  curl_setopt( $ch, CURLOPT_PROXYUSERPWD, $this->proxyConf[ 'user' ] . ':' . $this->proxyConf[ 'pass' ] );
876  }
877  }
878  curl_setopt( $ch, CURLOPT_FOLLOWLOCATION, true );
879  // headers will proceed the output, json_decode will fail below
880  curl_setopt( $ch, CURLOPT_HEADER, false );
881  curl_setopt( $ch, CURLOPT_RETURNTRANSFER, true );
882  curl_setopt( $ch, CURLOPT_TIMEOUT, 60 );
883 
884  // set user defined curl opts last for overriding
885  foreach( $this->curlOpts as $key => $value ) {
886  curl_setopt( $ch, constant( $key ), $value );
887  }
888 
889  $output = curl_exec( $ch );
890  // Check if any error occurred
891  if( curl_errno( $ch ) > 0 ) {
892  echo 'Curl error: ' . curl_error( $ch ) . "\n";
893  return [];
894  }
895  curl_close( $ch );
896  $json = json_decode( $output, true );
897  if( isset( $json[ 'msg' ] ) ) {
898  echo "signedRequest error: {$output}" . PHP_EOL;
899  }
900  $this->transfered += strlen( $output );
901  $this->requestCount++;
902  return $json;
903  }

1016  private function balanceData( array $array, $priceData ) {
1017  if( is_array( $priceData ) ) {
1018  $btc_value = $btc_total = 0.00;
1019  }
1020  $balances = [];
1021  if( empty( $array ) || empty( $array[ 'balances' ] ) ) {
1022  echo "balanceData error: Please make sure your system time is synchronized, or pass the useServerTime option." . PHP_EOL;
1023  return [];
1024  }
1025  foreach( $array[ 'balances' ] as $obj ) {
1026  $asset = $obj[ 'asset' ];
1027  $balances[ $asset ] = [
1028  "available" => $obj[ 'free' ],
1029  "onOrder" => $obj[ 'locked' ],
1030  "btcValue" => 0.00000000,
1031  "btcTotal" => 0.00000000
1032  ];
1033  if( $priceData ) {
1034  if( $obj[ 'free' ] + $obj[ 'locked' ] < 0.00000001 )
1035  continue;
1036  if( $asset == 'BTC' ) {
1037  $balances[ $asset ][ 'btcValue' ] = $obj[ 'free' ];
1038  $balances[ $asset ][ 'btcTotal' ] = $obj[ 'free' ] + $obj[ 'locked' ];
1039  $btc_value += $obj[ 'free' ];
1040  $btc_total += $obj[ 'free' ] + $obj[ 'locked' ];
1041  continue;
1042  }
1043  $symbol = $asset . 'BTC';
1044  if( $symbol == 'BTCUSDT' ) {
1045  $btcValue = number_format( $obj[ 'free' ] / $priceData[ 'BTCUSDT' ], 8, '.', '' );
1046  $btcTotal = number_format( ( $obj[ 'free' ] + $obj[ 'locked' ] ) / $priceData[ 'BTCUSDT' ], 8, '.', '' );
1047  }
1048  elseif( !isset( $priceData[ $symbol ] ) ) {
