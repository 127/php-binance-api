519  public function getProxyUriString()
520  {
521  $uri = isset( $this->proxyConf['proto'] ) ? $this->proxyConf['proto'] : "http";
522  // https://curl.haxx.se/libcurl/c/CURLOPT_PROXY.html
523  $supportedProxyProtocols = array('http', 'https', 'socks4', 'socks4a', 'socks5', 'socks5h');
524  if(in_array($uri, $supportedProxyProtocols) == false) {
525  die("Unknown proxy protocol '" . $this->proxyConf['proto'] . "', supported protocols are " . implode(", ",$supportedProxyProtocols) . "\n");
526  }
527  $uri .= "://";
528  $uri .= isset( $this->proxyConf['address'] ) ? $this->proxyConf['address'] : "localhost";
529  if( isset( $this->proxyConf['address'] ) == false ) echo "warning: proxy address not set defaulting to localhost\n";
530  $uri .= ":";
531  $uri .= isset( $this->proxyConf['port'] ) ? $this->proxyConf['port'] : "1080";
532  if( isset( $this->proxyConf['address'] ) == false ) echo "warning: proxy port not set defaulting to 1080\n";
533  return $uri;
534  }

571  private function httpRequest($url, $method = "GET", $params = [], $signed = false) {
572  // is cURL installed yet?
573  if (!function_exists('curl_init')) {
574  die('Sorry cURL is not installed!');
575  }
576  $ch = curl_init();
577  curl_setopt($ch, CURLOPT_VERBOSE, $this->httpDebug);
578  $query = http_build_query($params, '', '&');
579  // signed with params
580  if($signed == true) {
581  if(empty($this->api_key) ) die("signedRequest error: API Key not set!");
582  if(empty($this->api_secret) ) die("signedRequest error: API Secret not set!");
583  $base = $this->base;
584  $ts = (microtime(true)*1000) + $this->info['timeOffset'];
585  $params['timestamp'] = number_format($ts,0,'.','');
586  if(isset($params['wapi'])) {
587  unset($params['wapi']);
588  $base = $this->wapi;
589  }
590  $query = http_build_query($params, '', '&');
591  $signature = hash_hmac('sha256', $query, $this->api_secret);
592  $endpoint = $base.$url.'?'.$query.'&signature='.$signature;
593  curl_setopt($ch, CURLOPT_URL, $endpoint);
594  curl_setopt($ch, CURLOPT_HTTPHEADER, array('X-MBX-APIKEY: ' . $this->api_key));
595  }
596  // params so buildquery string and append to url
597  else if(count($params)>0){
598  curl_setopt($ch, CURLOPT_URL, $this->base.$url.'?'.$query);
599  }
600  // no params so just the base url
601  else {
602  curl_setopt($ch, CURLOPT_URL, $this->base.$url);
603  curl_setopt($ch, CURLOPT_HTTPHEADER, array('X-MBX-APIKEY: ' . $this->api_key));
604  }
605  curl_setopt($ch, CURLOPT_USERAGENT, "User-Agent: Mozilla/4.0 (compatible; PHP Binance API)");
606  // Post and postfields
607  if($method == "POST") {
608  curl_setopt($ch, CURLOPT_POST, true);
609  //curl_setopt($ch, CURLOPT_POSTFIELDS, $query);
610  }
611  // Delete Method
612  if($method == "DELETE") {
613  curl_setopt($ch, CURLOPT_CUSTOMREQUEST, $method);
614  }
615  // proxy settings
616  if(is_array($this->proxyConf)) {
617  curl_setopt($ch, CURLOPT_PROXY, $this->getProxyUriString());
618  if(isset($this->proxyConf['user']) && isset($this->proxyConf['pass'])) {
619  curl_setopt($ch, CURLOPT_PROXYUSERPWD, $this->proxyConf['user'] . ':' . $this->proxyConf['pass']);
620  }
621  }
622  curl_setopt($ch, CURLOPT_FOLLOWLOCATION, true);
623  // headers will proceed the output, json_decode will fail below
624  curl_setopt($ch, CURLOPT_HEADER, false);
625  curl_setopt($ch, CURLOPT_RETURNTRANSFER, true);
626  curl_setopt($ch, CURLOPT_TIMEOUT, 60);
627  $output = curl_exec($ch);
628  // Check if any error occurred
629  if(curl_errno($ch) > 0) {
630  echo 'Curl error: ' . curl_error($ch) . "\n";
631  return [];
632  }
633  curl_close($ch);
634  $json = json_decode($output, true);
635  if(isset($json['msg'])) {
636  echo "signedRequest error: {$output}".PHP_EOL;
637  }
638  $this->transfered += strlen( $output );
639  $this->requestCount++;
640  return $json;
641  }

726  private function balanceData($array, $priceData = false) {
727  if ( $priceData ) $btc_value = $btc_total = 0.00;
728  $balances = [];
729  if ( empty($array) || empty($array['balances']) ) {
730  echo "balanceData error: Please make sure your system time is synchronized, or pass the useServerTime option.".PHP_EOL;
731  return [];
732  }
733  foreach ( $array['balances'] as $obj ) {
734  $asset = $obj['asset'];
735  $balances[$asset] = ["available"=>$obj['free'], "onOrder"=>$obj['locked'], "btcValue"=>0.00000000, "btcTotal"=>0.00000000];
736  if ( $priceData ) {
737  if ( $obj['free'] + $obj['locked'] < 0.00000001 ) continue;
738  if ( $asset == 'BTC' ) {
739  $balances[$asset]['btcValue'] = $obj['free'];
740  $balances[$asset]['btcTotal'] = $obj['free'] + $obj['locked'];
741  $btc_value+= $obj['free'];
742  $btc_total+= $obj['free'] + $obj['locked'];
743  continue;
744  }
745  $symbol = $asset.'BTC';
746  if ( $symbol == 'USDTBTC' ) {
747  $btcValue = number_format($obj['free'] / $priceData['BTCUSDT'],8,'.','');
748  $btcTotal = number_format(($obj['free'] + $obj['locked']) / $priceData['BTCUSDT'],8,'.','');
749  } elseif ( !isset($priceData[$symbol]) ) {
750  $btcValue = $btcTotal = 0;
751  } else {
752  $btcValue = number_format($obj['free'] * $priceData[$symbol],8,'.','');
753  $btcTotal = number_format(($obj['free'] + $obj['locked']) * $priceData[$symbol],8,'.','');
754  }
755  $balances[$asset]['btcValue'] = $btcValue;
756  $balances[$asset]['btcTotal'] = $btcTotal;
757  $btc_value+= $btcValue;
758  $btc_total+= $btcTotal;
759  }
760  }
761  if ( $priceData ) {
762  uasort($balances, function($a, $b) { return $a['btcValue'] < $b['btcValue']; });
763  $this->btc_value = $btc_value;
764  $this->btc_total = $btc_total;
765  }
766  return $balances;
767  }
