768  private function httpRequest( string $url, string $method = "GET", array $params = [], bool $signed = false ) {
769  if( function_exists( 'curl_init' ) == false ) {
770  throw new \Exception( "Sorry cURL is not installed!" );
771  }
772 
773  $ch = curl_init();
774  curl_setopt( $ch, CURLOPT_VERBOSE, $this->httpDebug );
775  $query = http_build_query( $params, '', '&' );
776 
777  // signed with params
778  if( $signed == true ) {
779  if( empty( $this->api_key ) )
780  throw new \Exception( "signedRequest error: API Key not set!" );
781  if( empty( $this->api_secret ) )
782  throw new \Exception( "signedRequest error: API Secret not set!" );
783  $base = $this->base;
784  $ts = ( microtime( true ) * 1000 ) + $this->info[ 'timeOffset' ];
785  $params[ 'timestamp' ] = number_format( $ts, 0, '.', '' );
786  if( isset( $params[ 'wapi' ] ) ) {
787  unset( $params[ 'wapi' ] );
788  $base = $this->wapi;
789  }
790  $query = http_build_query( $params, '', '&' );
791  $signature = hash_hmac( 'sha256', $query, $this->api_secret );
792  $endpoint = $base . $url . '?' . $query . '&signature=' . $signature;
793  curl_setopt( $ch, CURLOPT_URL, $endpoint );
794  curl_setopt( $ch, CURLOPT_HTTPHEADER, array(
795  'X-MBX-APIKEY: ' . $this->api_key
796  ) );
797  }
798  // params so buildquery string and append to url
799  else if( count( $params ) > 0 ) {
800  curl_setopt( $ch, CURLOPT_URL, $this->base . $url . '?' . $query );
801  }
802  // no params so just the base url
803  else {
804  curl_setopt( $ch, CURLOPT_URL, $this->base . $url );
805  curl_setopt( $ch, CURLOPT_HTTPHEADER, array(
806  'X-MBX-APIKEY: ' . $this->api_key
807  ) );
808  }
809  curl_setopt( $ch, CURLOPT_USERAGENT, "User-Agent: Mozilla/4.0 (compatible; PHP Binance API)" );
810  // Post and postfields
811  if( $method == "POST" ) {
812  curl_setopt( $ch, CURLOPT_POST, true );
813  // curl_setopt($ch, CURLOPT_POSTFIELDS, $query);
814  }
815  // Delete Method
816  if( $method == "DELETE" ) {
817  curl_setopt( $ch, CURLOPT_CUSTOMREQUEST, $method );
818  }
819  // proxy settings
820  if( is_array( $this->proxyConf ) ) {
821  curl_setopt( $ch, CURLOPT_PROXY, $this->getProxyUriString() );
822  if( isset( $this->proxyConf[ 'user' ] ) && isset( $this->proxyConf[ 'pass' ] ) ) {
823  curl_setopt( $ch, CURLOPT_PROXYUSERPWD, $this->proxyConf[ 'user' ] . ':' . $this->proxyConf[ 'pass' ] );
824  }
825  }
826  curl_setopt( $ch, CURLOPT_FOLLOWLOCATION, true );
827  // headers will proceed the output, json_decode will fail below
828  curl_setopt( $ch, CURLOPT_HEADER, false );
829  curl_setopt( $ch, CURLOPT_RETURNTRANSFER, true );
830  curl_setopt( $ch, CURLOPT_TIMEOUT, 60 );
831  $output = curl_exec( $ch );
832  // Check if any error occurred
833  if( curl_errno( $ch ) > 0 ) {
834  echo 'Curl error: ' . curl_error( $ch ) . "\n";
835  return [];
836  }
837  curl_close( $ch );
838  $json = json_decode( $output, true );
839  if( isset( $json[ 'msg' ] ) ) {
840  echo "signedRequest error: {$output}" . PHP_EOL;
841  }
842  $this->transfered += strlen( $output );
843  $this->requestCount++;
844  return $json;
845  }

958  private function balanceData( array $array, $priceData ) {
959  if( is_array( $priceData ) ) {
960  $btc_value = $btc_total = 0.00;
961  }
962  $balances = [];
963  if( empty( $array ) || empty( $array[ 'balances' ] ) ) {
964  echo "balanceData error: Please make sure your system time is synchronized, or pass the useServerTime option." . PHP_EOL;
965  return [];
966  }
967  foreach( $array[ 'balances' ] as $obj ) {
968  $asset = $obj[ 'asset' ];
969  $balances[ $asset ] = [
970  "available" => $obj[ 'free' ],
971  "onOrder" => $obj[ 'locked' ],
972  "btcValue" => 0.00000000,
973  "btcTotal" => 0.00000000
974  ];
975  if( $priceData ) {
976  if( $obj[ 'free' ] + $obj[ 'locked' ] < 0.00000001 )
977  continue;
978  if( $asset == 'BTC' ) {
979  $balances[ $asset ][ 'btcValue' ] = $obj[ 'free' ];
980  $balances[ $asset ][ 'btcTotal' ] = $obj[ 'free' ] + $obj[ 'locked' ];
981  $btc_value += $obj[ 'free' ];
982  $btc_total += $obj[ 'free' ] + $obj[ 'locked' ];
983  continue;
984  }
985  $symbol = $asset . 'BTC';
986  if( $symbol == 'USDTBTC' ) {
987  $btcValue = number_format( $obj[ 'free' ] / $priceData[ 'BTCUSDT' ], 8, '.', '' );
988  $btcTotal = number_format( ( $obj[ 'free' ] + $obj[ 'locked' ] ) / $priceData[ 'BTCUSDT' ], 8, '.', '' );
989  }
990  elseif( !isset( $priceData[ $symbol ] ) ) {
991  $btcValue = $btcTotal = 0;
992  }
993  else {
994  $btcValue = number_format( $obj[ 'free' ] * $priceData[ $symbol ], 8, '.', '' );
995  $btcTotal = number_format( ( $obj[ 'free' ] + $obj[ 'locked' ] ) * $priceData[ $symbol ], 8, '.', '' );
996  }
997  $balances[ $asset ][ 'btcValue' ] = $btcValue;
998  $balances[ $asset ][ 'btcTotal' ] = $btcTotal;
999  $btc_value += $btcValue;
1000  $btc_total += $btcTotal;
1001  }
1002  }
1003  if( is_array( $priceData ) ) {
1004  uasort( $balances, function ( $a, $b ) {
1005  return $a[ 'btcValue' ] < $b[ 'btcValue' ];
1006  } );
1007  $this->btc_value = $btc_value;
1008  $this->btc_total = $btc_total;
1009  }
1010  return $balances;
1011  }
